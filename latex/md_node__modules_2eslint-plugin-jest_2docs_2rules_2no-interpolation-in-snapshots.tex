 Prevents the use of string interpolations in snapshots.\hypertarget{md_node__modules_2eslint-plugin-jest_2docs_2rules_2no-interpolation-in-snapshots_autotoc_md11410}{}\doxysection{\texorpdfstring{Rule Details}{Rule Details}}\label{md_node__modules_2eslint-plugin-jest_2docs_2rules_2no-interpolation-in-snapshots_autotoc_md11410}
Interpolation prevents snapshots from being updated. Instead, properties should be overloaded with a matcher by using \href{https://jestjs.io/docs/en/snapshot-testing\#property-matchers}{\texttt{ property matchers}}.

Examples of {\bfseries{incorrect}} code for this rule\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{expect(something).toMatchInlineSnapshot(}
\DoxyCodeLine{\ \ \`{}Object\ \{}
\DoxyCodeLine{\ \ \ \ property:\ \$\{interpolated\}}
\DoxyCodeLine{\ \ \}`,}
\DoxyCodeLine{);}
\DoxyCodeLine{}
\DoxyCodeLine{expect(something).toMatchInlineSnapshot(}
\DoxyCodeLine{\ \ \{\ other:\ expect.any(Number)\ \},}
\DoxyCodeLine{\ \ \`{}Object\ \{}
\DoxyCodeLine{\ \ \ \ other:\ Any<Number>,}
\DoxyCodeLine{\ \ \ \ property:\ \$\{interpolated\}}
\DoxyCodeLine{\ \ \}`,}
\DoxyCodeLine{);}
\DoxyCodeLine{}
\DoxyCodeLine{expect(errorThrowingFunction).toThrowErrorMatchingInlineSnapshot(}
\DoxyCodeLine{\ \ \`{}\$\{interpolated\}`,}
\DoxyCodeLine{);}

\end{DoxyCode}

\DoxyCodeLine{\ \ \ \ property:\ 1}
\DoxyCodeLine{\ \ \}`,}
\DoxyCodeLine{);}
\DoxyCodeLine{}

\DoxyCodeLine{\ \ \{\ property:\ expect.any(Date)\ \},}
\DoxyCodeLine{\ \ \`{}Object\ \{}
\DoxyCodeLine{\ \ \ \ property:\ Any<Date>}
\DoxyCodeLine{\ \ \}`,}
\DoxyCodeLine{);}
\DoxyCodeLine{}
\DoxyCodeLine{expect(errorThrowingFunction).toThrowErrorMatchingInlineSnapshot();}
\DoxyCodeLine{}
\DoxyCodeLine{expect(errorThrowingFunction).toThrowErrorMatchingInlineSnapshot(}
\DoxyCodeLine{\ \ \`{}Error\ Message`,}
\DoxyCodeLine{);}

\end{DoxyCode}
