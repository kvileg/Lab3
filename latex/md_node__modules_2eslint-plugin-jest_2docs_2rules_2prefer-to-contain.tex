 In order to have a better failure message, {\ttfamily to\+Contain()} should be used upon asserting expectations on an array containing an object.\hypertarget{md_node__modules_2eslint-plugin-jest_2docs_2rules_2prefer-to-contain_autotoc_md11467}{}\doxysection{\texorpdfstring{Rule details}{Rule details}}\label{md_node__modules_2eslint-plugin-jest_2docs_2rules_2prefer-to-contain_autotoc_md11467}
This rule triggers a warning if {\ttfamily to\+Be()}, {\ttfamily to\+Equal()} or {\ttfamily to\+Strict\+Equal()} is used to assert object inclusion in an array


\begin{DoxyCode}{0}
\DoxyCodeLine{expect(a.includes(b)).toBe(true);}

\end{DoxyCode}



\begin{DoxyCode}{0}
\DoxyCodeLine{expect(a.includes(b)).not.toBe(true);}

\end{DoxyCode}



\begin{DoxyCode}{0}
\DoxyCodeLine{expect(a.includes(b)).toBe(false);}

\end{DoxyCode}
\hypertarget{md_node__modules_2eslint-plugin-jest_2docs_2rules_2prefer-to-contain_autotoc_md11468}{}\doxysubsection{\texorpdfstring{Default configuration}{Default configuration}}\label{md_node__modules_2eslint-plugin-jest_2docs_2rules_2prefer-to-contain_autotoc_md11468}
The following patterns are considered warnings\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{expect(a.includes(b)).toBe(true);}
\DoxyCodeLine{}
\DoxyCodeLine{expect(a.includes(b)).not.toBe(true);}
\DoxyCodeLine{}
\DoxyCodeLine{expect(a.includes(b)).toBe(false);}
\DoxyCodeLine{}
\DoxyCodeLine{expect(a.includes(b)).toEqual(true);}
\DoxyCodeLine{}
\DoxyCodeLine{expect(a.includes(b)).toStrictEqual(true);}

\end{DoxyCode}


The following patterns are not considered warnings\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{expect(a).toContain(b);}
\DoxyCodeLine{}
\DoxyCodeLine{expect(a).not.toContain(b);}

\end{DoxyCode}
