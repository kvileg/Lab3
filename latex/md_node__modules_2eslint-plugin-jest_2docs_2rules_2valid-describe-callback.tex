 Using an improper {\ttfamily describe()} callback function can lead to unexpected test errors.\hypertarget{md_node__modules_2eslint-plugin-jest_2docs_2rules_2valid-describe-callback_autotoc_md11489}{}\doxysection{\texorpdfstring{Rule Details}{Rule Details}}\label{md_node__modules_2eslint-plugin-jest_2docs_2rules_2valid-describe-callback_autotoc_md11489}
This rule validates that the second parameter of a {\ttfamily describe()} function is a callback function. This callback function\+:


\begin{DoxyItemize}
\item should not be \href{https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Statements/async_function}{\texttt{ async}}
\item should not contain any parameters
\item should not contain any {\ttfamily return} statements
\end{DoxyItemize}

The following {\ttfamily describe} function aliases are also validated\+:


\begin{DoxyItemize}
\item {\ttfamily describe}
\item {\ttfamily describe.\+only}
\item {\ttfamily describe.\+skip}
\item {\ttfamily fdescribe}
\item {\ttfamily xdescribe}
\end{DoxyItemize}

The following patterns are considered warnings\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{//\ Async\ callback\ functions\ are\ not\ allowed}
\DoxyCodeLine{describe('myFunction()',\ async\ ()\ =>\ \{}
\DoxyCodeLine{\ \ //\ ...}
\DoxyCodeLine{\});}
\DoxyCodeLine{}
\DoxyCodeLine{//\ Callback\ function\ parameters\ are\ not\ allowed}
\DoxyCodeLine{describe('myFunction()',\ done\ =>\ \{}
\DoxyCodeLine{\ \ //\ ...}
\DoxyCodeLine{\});}
\DoxyCodeLine{}
\DoxyCodeLine{//}
\DoxyCodeLine{describe('myFunction',\ ()\ =>\ \{}
\DoxyCodeLine{\ \ //\ No\ return\ statements\ are\ allowed\ in\ block\ of\ a\ callback\ function}
\DoxyCodeLine{\ \ return\ Promise.resolve().then(()\ =>\ \{}
\DoxyCodeLine{\ \ \ \ it('breaks',\ ()\ =>\ \{}
\DoxyCodeLine{\ \ \ \ \ \ throw\ new\ Error('Fail');}
\DoxyCodeLine{\ \ \ \ \});}
\DoxyCodeLine{\ \ \});}
\DoxyCodeLine{\});}
\DoxyCodeLine{}
\DoxyCodeLine{//\ Returning\ a\ value\ from\ a\ describe\ block\ is\ not\ allowed}
\DoxyCodeLine{describe('myFunction',\ ()\ =>}
\DoxyCodeLine{\ \ it('returns\ a\ truthy\ value',\ ()\ =>\ \{}
\DoxyCodeLine{\ \ \ \ expect(myFunction()).toBeTruthy();}
\DoxyCodeLine{\ \ \}));}

\end{DoxyCode}


The following patterns are not considered warnings\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{describe('myFunction()',\ ()\ =>\ \{}
\DoxyCodeLine{\ \ it('returns\ a\ truthy\ value',\ ()\ =>\ \{}
\DoxyCodeLine{\ \ \ \ expect(myFunction()).toBeTruthy();}
\DoxyCodeLine{\ \ \});}
\DoxyCodeLine{\});}

\end{DoxyCode}
